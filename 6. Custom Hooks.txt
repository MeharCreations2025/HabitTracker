// hooks/useHabits.ts
import { useState, useEffect } from 'react';
import { Habit, DailyProgress } from '../types';

export const useHabits = () => {
  const [habits, setHabits] = useState<Habit[]>([]);
  const [progress, setProgress] = useState<DailyProgress>({
    date: new Date().toISOString().split('T')[0],
    habits: {},
    prayers: [],
    socialMediaTime: 0,
    socialMediaUseful: false,
    dailyScore: 0,
    reflection: null
  });

  useEffect(() => {
    // Load habits from localStorage or initialize defaults
    const savedHabits = localStorage.getItem('habits');
    if (savedHabits) {
      setHabits(JSON.parse(savedHabits));
    } else {
      initializeDefaultHabits();
    }

    // Load today's progress
    const today = new Date().toISOString().split('T')[0];
    const savedProgress = localStorage.getItem(`progress-${today}`);
    if (savedProgress) {
      setProgress(JSON.parse(savedProgress));
    }
  }, []);

  const initializeDefaultHabits = () => {
    const defaultHabits: Habit[] = [
      // After Fajr
      { id: '1', name: 'Wazifa', category: 'wazifa', prayerPeriod: 'fajr', enabled: true, sundayEnabled: true, order: 1 },
      { id: '2', name: 'Quran Reading', category: 'quran', prayerPeriod: 'fajr', enabled: true, sundayEnabled: true, order: 2 },
      { id: '3', name: 'Heavy Breakfast', category: 'work', prayerPeriod: 'fajr', enabled: true, sundayEnabled: true, order: 3 },
      
      // Before/During Dhuhr
      { id: '4', name: 'Work', category: 'work', prayerPeriod: 'dhuhr', enabled: true, sundayEnabled: false, order: 1 },
      { id: '5', name: 'Money Progress Check', category: 'work', prayerPeriod: 'dhuhr', enabled: true, sundayEnabled: false, order: 2 },
      { id: '6', name: 'Learning', category: 'work', prayerPeriod: 'dhuhr', enabled: true, sundayEnabled: false, order: 3 },
      { id: '7', name: 'Lunch', category: 'work', prayerPeriod: 'dhuhr', enabled: true, sundayEnabled: true, order: 4 },
      { id: '8', name: 'Exercise', category: 'exercise', prayerPeriod: 'dhuhr', enabled: true, sundayEnabled: false, order: 5 },
      
      // After Asr
      { id: '9', name: 'Outdoor Activity', category: 'exercise', prayerPeriod: 'asr', enabled: true, sundayEnabled: true, order: 1 },
      
      // After Maghrib
      { id: '10', name: 'Quran Reading', category: 'quran', prayerPeriod: 'maghrib', enabled: true, sundayEnabled: true, order: 1 },
      { id: '11', name: 'Dinner', category: 'work', prayerPeriod: 'maghrib', enabled: true, sundayEnabled: true, order: 2 },
      
      // After Isha
      { id: '12', name: 'Optional Running', category: 'exercise', prayerPeriod: 'isha', enabled: true, sundayEnabled: false, order: 1 },
      { id: '13', name: 'Family Time', category: 'social', prayerPeriod: 'isha', enabled: true, sundayEnabled: true, order: 2 },
      { id: '14', name: 'Social Media Check', category: 'social', prayerPeriod: 'isha', enabled: true, sundayEnabled: true, order: 3 },
      { id: '15', name: 'Sleep (10-11 PM)', category: 'work', prayerPeriod: 'isha', enabled: true, sundayEnabled: true, order: 4 },
    ];
    
    setHabits(defaultHabits);
    localStorage.setItem('habits', JSON.stringify(defaultHabits));
  };

  const updateHabit = (habitId: string, status: 'completed' | 'skipped') => {
    const newProgress = {
      ...progress,
      habits: {
        ...progress.habits,
        [habitId]: status
      }
    };
    
    setProgress(newProgress);
    localStorage.setItem(`progress-${progress.date}`, JSON.stringify(newProgress));
  };

  const getDailyScore = () => {
    const totalHabits = habits.filter(h => h.enabled).length;
    const completedHabits = Object.values(progress.habits).filter(status => status === 'completed').length;
    return totalHabits > 0 ? Math.round((completedHabits / totalHabits) * 100) : 0;
  };

  return {
    habits,
    progress,
    updateHabit,
    getDailyScore
  };
};

// hooks/usePrayerTimes.ts
import { useState, useEffect } from 'react';

export interface PrayerTimes {
  fajr: string;
  dhuhr: string;
  asr: string;
  maghrib: string;
  isha: string;
}

export const usePrayerTimes = () => {
  const [prayerTimes, setPrayerTimes] = useState<PrayerTimes | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchPrayerTimes();
  }, []);

  const fetchPrayerTimes = async () => {
    try {
      // Try to get user's location
      if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(
          async (position) => {
            const { latitude, longitude } = position.coords;
            await getPrayerTimesForLocation(latitude, longitude);
          },
          () => {
            // Fallback to default location (Mecca)
            getPrayerTimesForLocation(21.4225, 39.8262);
          }
        );
      } else {
        // Fallback to default location
        getPrayerTimesForLocation(21.4225, 39.8262);
      }
    } catch (error) {
      console.error('Error fetching prayer times:', error);
      setLoading(false);
    }
  };

  const getPrayerTimesForLocation = async (lat: number, lng: number) => {
    try {
      const today = new Date();
      const response = await fetch(
        `https://api.aladhan.com/v1/timings/${today.getDate()}-${today.getMonth() + 1}-${today.getFullYear()}?latitude=${lat}&longitude=${lng}&method=2`
      );
      
      const data = await response.json();
      
      if (data.code === 200) {
        const timings = data.data.timings;
        setPrayerTimes({
          fajr: timings.Fajr,
          dhuhr: timings.Dhuhr,
          asr: timings.Asr,
          maghrib: timings.Maghrib,
          isha: timings.Isha
        });
      }
    } catch (error) {
      console.error('Error fetching prayer times:', error);
    } finally {
      setLoading(false);
    }
  };

  return { prayerTimes, loading };
};
